   kubectl get crd inferenceservices.serving.trin.io

4. 创建示例 InferenceService：
   kubectl apply -f - <<EOF

   kind: InferenceService
   metadata:
     name: test-model
     namespace: {{ .Release.Namespace }}
   spec:
     modelName: "test-model"
     engine: "mock"
     replicas: 1
   EOF

5. 查看 InferenceService 状态：
   kubectl get inferenceservices -n {{ .Release.Namespace }}
   kubectl describe inferenceservice test-model -n {{ .Release.Namespace }}
